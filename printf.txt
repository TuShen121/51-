#include<reg52.h>  
#include<stdio.h>  
	
void uart(void)                 //串口通信初始化  
{  
	TMOD = 0X20;//打开定时器
	TH1 = 0XFD;
	TL1 = 0XFD;
	TR1 = 1;
	SCON = 0X40;
	TI = 1;
}                                 
void main(void)  
{  
	int i=12345;
    uart();  	
    while(1)  
        {     
            printf("123%d\n",i);
        }  
}
